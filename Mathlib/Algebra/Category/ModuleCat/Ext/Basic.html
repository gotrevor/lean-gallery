    theorem CategoryTheory.Abelian.Ext.postcomp_smul_id_mono_iff {R : Type u} [CommRing R] [Small.{v, u} R] {M N : ModuleCat R} (r : R) (i : ℕ) :

    r • 𝟙 M induces a monomorphism in Ext M N n if and only if scalar multiplication by r is faithful on Ext M N n.
